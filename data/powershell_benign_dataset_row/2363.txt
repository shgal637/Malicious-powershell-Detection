function Get-ClientsUnderNoSite{[CmdletBinding()]param ()try{$dcs = ((Get-ADForest).Domains | foreach { (Get-ADDomainController -Server $_ -Filter *) }).HostNameforeach ($d in $dcs){$output = @{'DomainController' = $d}$clients = Select-String -Pattern 'NO_CLIENT_SITE: (.*) \d' -Path "\\$d\c$\windows\debug\netlogon.log" | foreach {$_.Matches.Groups[1].Value} | Group-Objectif ($clients){$clients | foreach {$output.Client = $_.Name[pscustomobject]$output}}}}catch{Write-Error -Message $_.Exception.Message}}
