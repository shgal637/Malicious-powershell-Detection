Register-PSFConfigValidation -Name "consolecolor" -ScriptBlock {Param ($Value)$Result = New-Object PSObject -Property @{Success = $TrueValue   = $nullMessage = ""}try { [System.ConsoleColor]$color = $Value }catch{$Result.Message = "Not a console color: $Value"$Result.Success = $Falsereturn $Result}$Result.Value = $colorreturn $Result}
