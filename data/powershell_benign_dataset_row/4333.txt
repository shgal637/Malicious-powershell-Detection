function Get-PublishLocation{[CmdletBinding()]Param([Parameter()][String]$Location)$PublishLocation = $nullif($Location){if(Microsoft.PowerShell.Management\Test-Path -Path $Location){$PublishLocation = $Location}else{$tempPublishLocation = $nullif($Location.EndsWith('/api/v2', [System.StringComparison]::OrdinalIgnoreCase)){$tempPublishLocation = $Location + '/package/'}elseif($Location.EndsWith('/api/v2/', [System.StringComparison]::OrdinalIgnoreCase)){$tempPublishLocation = $Location + 'package/'}if($tempPublishLocation){$PublishLocation = $tempPublishLocation}}}return $PublishLocation}
