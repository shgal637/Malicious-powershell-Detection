Register-PSFConfigValidation -Name "integer" -ScriptBlock {Param ($Value)$Result = New-Object PSOBject -Property @{Success = $TrueValue = $nullMessage = ""}try { [int]$number = $Value }catch{$Result.Message = "Not an integer: $Value"$Result.Success = $Falsereturn $Result}$Result.Value = $numberreturn $Result}
