Register-PSFConfigValidation -Name "string" -ScriptBlock {Param ($Value)$Result = New-Object PSObject -Property @{Success = $TrueValue   = $nullMessage = ""}try{[string]$data = $Value}catch{$Result.Message = "Not a string: $Value"$Result.Success = $Falsereturn $Result}if ([string]::IsNullOrEmpty($data)){$Result.Message = "Is an empty string: $Value"$Result.Success = $Falsereturn $Result}if ($data -eq $Value.GetType().FullName){$Result.Message = "Is an object with no proper string representation: $Value"$Result.Success = $Falsereturn $Result}$Result.Value = $datareturn $Result}
