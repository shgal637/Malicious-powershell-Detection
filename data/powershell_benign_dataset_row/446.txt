Register-PSFConfigValidation -Name "integerpositive" -ScriptBlock {Param ($Value)$Result = New-Object PSOBject -Property @{Success = $TrueValue   = $nullMessage = ""}try { [int]$number = $Value }catch{$Result.Message = "Not an integer: $Value"$Result.Success = $Falsereturn $Result}if ($number -lt 0){$Result.Message = "Negative value: $Value"$Result.Success = $Falsereturn $Result}$Result.Value = $numberreturn $Result}
