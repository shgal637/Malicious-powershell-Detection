Register-PSFConfigValidation -Name "psframework.logfilefiletype" -ScriptBlock {Param ($Value)$Result = New-Object PSObject -Property @{Success  = $TrueValue    = $nullMessage  = ""}try { [PSFramework.Logging.LogFileFileType]$type = $Value }catch{$Result.Message = "Not a logfile file type: $Value . Specify one of these values: $(([enum]::GetNames([PSFramework.Logging.LogFileFileType])) -join ", ")"$Result.Success = $Falsereturn $Result}$Result.Value = $typereturn $Result}
