function Test-CRegistryKeyValue{[CmdletBinding()]param([Parameter(Mandatory=$true)][string]$Path,[Parameter(Mandatory=$true)][string]$Name)Set-StrictMode -Version 'Latest'Use-CallerPreference -Cmdlet $PSCmdlet -Session $ExecutionContext.SessionStateif( -not (Test-Path -Path $Path -PathType Container) ){return $false}$properties = Get-ItemProperty -Path $Path if( -not $properties ){return $false}$member = Get-Member -InputObject $properties -Name $Nameif( $member ){return $true}else{return $false}}
